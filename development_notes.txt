To fix "AttributeError: module 'sys' has no attribute 'argv'", copy the following three lines to /usr/local/lib/python3.6/dist-packages/tensorflow_core/python/platform/flags.py
	import sys
	if not hasattr(sys, 'argv'):
    	sys.argv  = ['']
